ll gcd(ll a, ll b) { return b ? gcd(b, a % b) : a; }                    // #2
ll qpow(ll a, ll b, ll m) { ll r=1; while(b){ if(b&1)r=r*a%m; a=a*a%m; b>>=1; } return r; }  // #7
ll exgcd(ll a, ll b, ll &x, ll &y) { if(!b){x=1,y=0;return a;} ll g=exgcd(b,a%b,y,x); y-=a/b*x; return g; } // #8
ll inv(ll a, ll p) { return qpow(a, p-2, p); }                           // #9（p为素数时）

## 筛法

```cpp
// 欧拉线性筛（素数 + 最小质因子）— #5
vector<int> primes; bool vis[N];
for (int i=2; i<=n; i++) {
    if (!vis[i]) primes.push_back(i);
    for (int p : primes) { if (i*p>n) break; vis[i*p]=1; if (i%p==0) break; }
}
```

## 组合数学

```cpp
ll C(ll n, ll k) { if(k>n-k)k=n-k; ll r=1; for(ll i=1;i<=k;i++) r=r*(n-i+1)/i; return r; }  // #29
// 卡特兰数递推 — #37:  C[i+1] = C[i]*(4*i+2)/(i+2)
// 第二类斯特林递推 — #38: S[i][j] = S[i-1][j-1] + j*S[i-1][j]
```

## 计算几何

const double eps = 1e-9;
int sgn(double x) { return (x>eps)-(x<-eps); }                           // #84
struct Pt { double x,y; Pt operator-(Pt b){return {x-b.x,y-b.y};} };
double cross(Pt a, Pt b) { return a.x*b.y - a.y*b.x; }                  // #84 叉积
